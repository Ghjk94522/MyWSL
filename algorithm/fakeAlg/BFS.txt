def BFS(v):
    Initialize an empty queue queNode;
    v.color := GRAY;
    v.dis := 0;
    queNode.ENQUE(v);
    while queNode != empty do:
        w := queNode.DEQUE();
        for each neighbour x of w do:
            if x.color = WHITE then:
                x.color = GRAY;
                x.parent := w;
                x.dis := w.dis+1;
                queNode.ENQUEUE(x);
        <processing of node w>;
        w.color := BLACK;
